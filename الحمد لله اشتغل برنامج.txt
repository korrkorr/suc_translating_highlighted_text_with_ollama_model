#!/bin/bash

# 1. الإعدادات
MODEL="translategemma:12b"

# 2. جلب النص المظلل (دعم كامل لـ Wayland و X11)
if [ "$XDG_SESSION_TYPE" = "wayland" ]; then
    TEXT=$(wl-paste --primary 2>/dev/null || wl-paste)
else
    TEXT=$(xclip -o)
fi

# إذا كان النص فارغاً
if [ -z "$TEXT" ]; then
    notify-send "Ollama" "لم يتم تظليل أي نص!"
    exit 1
fi

# 3. بناء الطلب بصيغة JSON صحيحة 100% باستخدام jq
# هذه الطريقة تعالج علامات التنصيص والرموز الغريبة تلقائياً
PAYLOAD=$(jq -n \
    --arg md "$MODEL" \
    --arg pr "Translate the following text to Arabic. Output only the translation. \n\nText: $TEXT" \
    '{model: $md, prompt: $pr, stream: false, options: {temperature: 0.2}}')

# 4. إرسال الطلب وحفظ الرد الكامل في متغير
RESPONSE_RAW=$(curl -s -X POST http://127.0.0.1:11434/api/generate -d "$PAYLOAD")

# 5. استخراج الترجمة أو الخطأ
TRANSLATION=$(echo "$RESPONSE_RAW" | jq -r '.response // empty')
ERROR_MSG=$(echo "$RESPONSE_RAW" | jq -r '.error // empty')

# 6. العرض النهائي
if [ -n "$TRANSLATION" ]; then
    # عرض الترجمة في نافذة
    echo -e "الترجمة (بواسطة $MODEL):\n\n$TRANSLATION" | zenity --text-info --title="Ollama Translate" --width=700 --height=500 --font="Ubuntu 14"
elif [ -n "$ERROR_MSG" ]; then
    zenity --error --text="خطأ من Ollama: $ERROR_MSG"
else
    # إذا كان الرد فارغاً تماماً، نعرض الرد الخام للتشخيص
    zenity --error --text="رد غير معروف. تأكد أن الموديل $MODEL محمل.\n\nالرد الخام: $RESPONSE_RAW"
fi

#!/bin/bash

# 1. تحديد النموذج الافتراضي (يمكنك تغييره هنا)
# إذا أردت استخدامه مباشرة دون سؤال، اجعل القيمة ثابتة
DEFAULT_MODEL="translategemma:12b"

# 2. الحصول على النص المظلل
if [ "$XDG_SESSION_TYPE" = "wayland" ]; then
    TEXT=$(wl-paste --primary)
    if [ -z "$TEXT" ]; then TEXT=$(wl-paste); fi
else
    TEXT=$(xclip -o)
fi

if [ -z "$TEXT" ]; then
    zenity --error --text="الرجاء تظليل نص أولاً!" --title="خطأ"
    exit 1
fi

# 3. جلب قائمة الموديلات المثبتة في Ollama (اختياري: إذا أردت اختيار موديل مختلف كل مرة)
# إذا كنت تريد استخدام الموديل الثابت دائماً، ابقِ السطر التالي معلقاً (#)
SELECTED_MODEL=$(ollama list | awk '{if(NR>1) print $1}' | zenity --list --title="اختر نموذج الترجمة" --column="الموديلات المتوفرة" --default-value="$DEFAULT_MODEL")

# إذا لم يتم اختيار موديل (إغلاق النافذة)، استخدم الافتراضي
if [ -z "$SELECTED_MODEL" ]; then
    SELECTED_MODEL=$DEFAULT_MODEL
fi

# 4. إرسال النص إلى Ollama
# ملاحظة: تم تحسين البرومبت ليتناسب مع نماذج الترجمة المتخصصة
RESPONSE=$(curl -s http://localhost:11434/api/generate -d '{
  "model": "'"$SELECTED_MODEL"'",
  "prompt": "Translate the following text to Arabic. Return only the translation:\n\n'"$TEXT"'",
  "stream": false
}' | jq -r '.response')

# 5. عرض النتيجة
if [ -z "$RESPONSE" ] || [ "$RESPONSE" == "null" ]; then
    zenity --error --text="فشل الاتصال بـ Ollama أو النموذج غير موجود."
else
    echo "$RESPONSE" | zenity --text-info --title="الترجمة بواسطة $SELECTED_MODEL" --width=600 --height=400 --font="Ubuntu 14" --editable
fi

# 1. تحديد النموذج الذي تملكه
SELECTED_MODEL="translategemma:12b"

# 2. الحصول على النص المظلل
if [ "$XDG_SESSION_TYPE" = "wayland" ]; then
    TEXT=$(wl-paste --primary)
    if [ -z "$TEXT" ]; then TEXT=$(wl-paste); fi
else
    TEXT=$(xclip -o)
fi

# التحقق من وجود نص
if [ -z "$TEXT" ]; then
    zenity --error --text="الرجاء تظليل نص أولاً!" --title="خطأ"
    exit 1
fi

# 3. إرسال النص إلى Ollama بطريقة آمنة (عبر تجميع الـ JSON بواسطة jq)
# هذه الطريقة تمنع حدوث أخطاء بسبب علامات التنصيص في النص الأصلي

    zenity --error --text="تعذر الحصول على رد من الموديل ($SELECTED_MODEL). تأكد أن Ollama يعمل." --title="خطأ في الاتصال"

المشكلة في النصوص الكبيرة غالباً ما تكون بسبب "نافذة السياق" (Context Window)، حيث يحدد Ollama افتراضياً عدداً معيناً من التوكنز (غالباً 2048 أو 4096)، وإذا زاد النص عن ذلك قد يتوقف أو يعطي نتائج ناقصة.

سأقوم بتعديل الكود لإضافة إعدادات متقدمة (Options) ترفع سعة الـ Context وتسمح بإنتاج ردود طويلة، مع إضافة تنبيه (Notification) يخبرك بأن الترجمة جارية لأن النصوص الكبيرة تأخذ وقتاً أطول للمعالجة.
الكود المطور للتعامل مع النصوص الكبيرة:
code Bash

#!/bin/bash

# 1. الإعدادات
SELECTED_MODEL="translategemma:12b"
CONTEXT_SIZE=8192  # حجم الذاكرة لاستيعاب نص كبير (Context Window)
MAX_PREDICT=4096   # أقصى عدد توكنز للرد (الترجمة)

# 2. الحصول على النص المظلل
if [ "$XDG_SESSION_TYPE" = "wayland" ]; then
    TEXT=$(wl-paste --primary)
    if [ -z "$TEXT" ]; then TEXT=$(wl-paste); fi
else
    TEXT=$(xclip -o)
fi

if [ -z "$TEXT" ]; then
    zenity --error --text="الرجاء تظليل نص أولاً!" --title="خطأ"
    exit 1
fi

# 3. إرسال تنبيه بسيط لتعرف أن المعالجة بدأت (مفيد للنصوص الكبيرة)
notify-send "Ollama" "بدأت عملية ترجمة نص طويل، يرجى الانتظار..." -i accessories-character-map

# 4. تجهيز الطلب بصيغة JSON مع بارامترات الحجم
# أضفنا num_ctx و num_predict داخل قسم options
JSON_DATA=$(jq -n \
  --arg model "$SELECTED_MODEL" \
  --arg prompt "Translate the following text to Arabic. Maintain the original meaning and formatting. Output ONLY the translation:\n\n$TEXT" \
  --argctx "$CONTEXT_SIZE" \
  --argpred "$MAX_PREDICT" \
  '{
    model: $model, 
    prompt: $prompt, 
    stream: false, 
    options: {
      num_ctx: ($argctx | tonumber),
      num_predict: ($argpred | tonumber),
      temperature: 0.3
    }
  }')

# 5. تنفيذ الطلب
RESPONSE=$(curl -s -X POST http://localhost:11434/api/generate -d "$JSON_DATA" | jq -r '.response' 2>/dev/null)

# 6. عرض النتيجة
if [ -z "$RESPONSE" ] || [ "$RESPONSE" == "null" ]; then
    zenity --error --text="فشل في معالجة النص. قد يكون النص ضخماً جداً أو الموديل توقف."
else
    # عرض النتيجة في نافذة قابلة للتمرير والقراءة بشكل مريح
    echo "$RESPONSE" | zenity --text-info --title="الترجمة بواسطة $SELECTED_MODEL" --width=800 --height=600 --font="Ubuntu 14"
fi

التغييرات الجوهرية في هذا الكود:

    num_ctx: 8192: قمنا بزيادة حجم "ذاكرة" الموديل. هذا يسمح له بقراءة نصوص طويلة جداً (آلاف الكلمات) دون أن ينسى البداية.

    num_predict: 4096: قمنا بزيادة طول الرد المسموح به، لضمان أن الترجمة لن تنقطع في المنتصف.

    temperature: 0.3: خفضنا درجة "الإبداع" لجعل الموديل أكثر دقة في الترجمة الحرفية والالتزام بالنص.

    notify-send: بما أن الموديل 12b ثقيل ويحتاج وقتاً لترجمة صفحات كاملة، ستظهر لك رسالة صغيرة في أعلى الشاشة بمجرد ضغط الاختصار لتطمئن أن السكربت يعمل.

    توسيع النافذة: جعلت نافذة النتائج (zenity) أكبر (800x600) لتناسب النصوص الطويلة.

PAYLOAD=$(jq -n \
    --arg md "$MODEL" \
    --arg pr "Translate the following text to Arabic. Output only the translation. \n\nText: $TEXT" \
    '{model: $md, prompt: $pr, stream: false, options: {temperature: 0.2}}')

# 4. إرسال الطلب وحفظ الرد الكامل في متغير
RESPONSE_RAW=$(curl -s -X POST http://127.0.0.1:11434/api/generate -d "$PAYLOAD")

# 5. استخراج الترجمة أو الخطأ
TRANSLATION=$(echo "$RESPONSE_RAW" | jq -r '.response // empty')
ERROR_MSG=$(echo "$RESPONSE_RAW" | jq -r '.error // empty')

# 6. العرض النهائي
if [ -n "$TRANSLATION" ]; then
    # عرض الترجمة في نافذة
    echo -e "الترجمة (بواسطة $MODEL):\n\n$TRANSLATION" | zenity --text-info --title="Ollama Translate" --width=700 --height=500 --font="Ubuntu 14"
elif [ -n "$ERROR_MSG" ]; then
    zenity --error --text="خطأ من Ollama: $ERROR_MSG"
else
    # إذا كان الرد فارغاً تماماً، نعرض الرد الخام للتشخيص
    zenity --error --text="رد غير معروف. تأكد أن الموديل $MODEL محمل.\n\nالرد الخام: $RESPONSE_RAW"
fi

لماذا هذا الكود سيحل مشكلتك؟

    JSON Safe: في المحاولات السابقة، كنا نضع $TEXT داخل علامات تنصيص يدوية، وهذا "ينفجر" برمجياً إذا كان النص يحتوي على أي رمز خاص. الآن jq -n --arg هو الذي يتولى تحويل النص إلى صيغة يفهمها السيرفر بأمان.

    التشخيص الذكي: إذا لم تنجح الترجمة، السكربت سيخبرك الآن هل الخطأ من السيرفر (Ollama) أم أن الرد جاء فارغاً.

    تبسيط الـ Prompt: جعلت الأوامر للموديل مباشرة جداً لضمان أسرع استجابة.

خطوات أخيرة للتأكد من كل شيء:

    تأكد من اسم الموديل: في الـ Terminal، نفذ ollama list وتأكد أن الاسم هو translategemma:12b بالضبط.

    تأكد من وجود الأدوات:
    code Bash

    sudo apt install jq curl zenity wl-clipboard xclip

    تجربة يدوية: بعد حفظ السكربت، ظلل كلمة "Desktop" مثلاً، ثم افتح الـ Terminal واكتب:
    code Bash

    bash ~/ollama_translate.sh

    إذا ظهرت نافذة فيها كلمة "سطح المكتب"، فالاختصار سيعمل معك فوراً.
